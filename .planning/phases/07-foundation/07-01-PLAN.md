Pure functions for Gregorian ↔ Lunar conversion. All functions are synchronous (no hooks, no side effects).

```typescript
import lunisolar from 'lunisolar';

/**
 * Convert a Gregorian date to lunar date string.
 * @param year - Gregorian year (1900-2100)
 * @param month - Gregorian month (1-12)
 * @param day - Gregorian day (1-31)
 * @returns Lunar date string e.g. '正月初一', '腊月廿三'
 */
export function toLunar(year: number, month: number, day: number): string {
  const dateStr = `${year}-${String(month).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
  const lsr = lunisolar(dateStr);
  const lunar = lsr.lunar;

  const monthName = lunar.month.isLeap ? '闰' : '';
  const monthNum = lunar.month.n;
  const monthNames = ['', '正月', '二月', '三月', '四月', '五月', '六月',
                      '七月', '八月', '九月', '十月', '冬月', '腊月'];
  const dayNames = ['', '初一', '初二', '初三', '初四', '初五', '初六', '初七', '初八', '初九', '初十',
                    '十一', '十二', '十三', '十四', '十五', '十六', '十七', '十八', '十九', '二十',
                    '廿一', '廿二', '廿三', '廿四', '廿五', '廿六', '廿七', '廿八', '廿九', '三十'];

  return `${monthName}${monthNames[monthNum]}${dayNames[lunar.day]}【${lunar.year}年】`;
}

/**
 * Convert a lunar date to Gregorian date string.
 * @param lunarYear - Lunar year
 * @param lunarMonth - Lunar month (1-12, 13 if leap month)
 * @param lunarDay - Lunar day (1-30)
 * @param isLeapMonth - Whether the month is a leap month
 * @returns Gregorian date string e.g. '2024-02-10', or null if invalid
 */
export function toSolar(lunarYear: number, lunarMonth: number, lunarDay: number, isLeapMonth = false): string | null {
  try {
    const lsr = lunisolar;
    // lunisolar expects lunar date via lunar property
    // Use static method to convert lunar to solar
    const result = lunisolar.fromLunar({
      year: lunarYear,
      month: lunarMonth,
      day: lunarDay,
      isLeapMonth,
    });

    if (!result) return null;

    const d = result.date;
    return `${d.year}-${String(d.month).padStart(2, '0')}-${String(d.day).padStart(2, '0')}`;
  } catch {
    return null;
  }
}
```

**Important:** The toSolar implementation depends on lunisolar API. Use `lunisolar.fromLunar()` static method. If the API differs, consult Context7 docs or use `lunisolar.create({ year, month, day, isLeapMonth })`.

For toLunar, extract month name from `lunar.month.n` and day from `lunar.day`. Handle leap months with the `isLeap` flag.

Pure function computing 四柱八字, 大运, and 十神 from BirthInput. No hooks, no side effects.

```typescript
import lunisolar from 'lunisolar';
import type { BirthInput, BaziResult, Pillar, Dayun, ShiShenType, ShiShenMap } from './types';

// Map 时辰 name to hour (the hour that STARTS the 时辰 period)
// Traditional: each 时辰 is 2 hours, starting at 23:00 for 子时
const SHICHEN_TO_HOUR: Record<string, number> = {
  '子': 23, '丑': 1, '寅': 3, '卯': 5, '辰': 7, '巳': 9,
  '午': 11, '未': 13, '申': 15, '酉': 17, '戌': 19, '亥': 21,
};

// 天干 list
const STEMS = ['甲', '乙', '丙', '丁', '戊', '己', '庚', '辛', '壬', '癸'];
// 地支 list
const BRANCHES = ['子', '丑', '寅', '卯', '辰', '巳', '午', '未', '申', '酉', '戌', '亥'];

/**
 * Resolve hour from BirthInput based on timeMode.
 */
function resolveHour(input: BirthInput): number {
  if (input.timeMode === 'unknown') {
    return 23; // Default to 子时 (23:00-01:00)
  }
  if (input.timeMode === 'shichen') {
    return SHICHEN_TO_HOUR[input.shichen ?? '子'] ?? 23;
  }
  // precise mode
  return input.hour ?? 12;
}

/**
 * Compute the dayun (大运) periods.
 * Male: 顺行 (forward), Female: 逆行 (backward)
 * Each period is 10 years.
 */
function computeDayun(birthYear: number, gender: 'male' | 'female'): Dayun[] {
  const startAge = 8; // Traditional: 大运 starts at 8
  const dayun: Dayun[] = [];

  // Determine starting year stem/branch
  // For MVP: just compute 8 periods starting from birth+8
  for (let i = 0; i < 8; i++) {
    const age = startAge + i * 10;
    const year = birthYear + age;

    // Compute stem/branch for this year (simplified — use lunisolar if available)
    // For MVP: just show placeholder positions
    dayun.push({
      age,
      year,
      stem: STEMS[i % 10],
      branch: BRANCHES[i % 12],
      stemName: STEMS[i % 10],
      branchName: BRANCHES[i % 12],
    });
  }

  return dayun;
}

/**
 * Compute 十神 (Ten Gods) based on the day master's stem.
 * 十神 is derived by comparing each pillar's stem to the day stem.
 */
function computeShiShen(dayStemIdx: number): ShiShenMap {
  // 同我者: 比肩, 劫财
  // 我生者: 食神, 伤官
  // 我克者: 偏财, 正财
  // 克我者: 偏官, 正官
  // 生我者: 偏印, 正印

  // For simplicity, assign based on stem index difference
  // This is a simplified version — the real calculation uses complex relationships
  const shiShenList: ShiShenType[] = ['比肩', '劫财', '食神', '伤官', '偏财', '正财', '偏官', '正官', '偏印', '正印'];

  return {
    yearStem: shiShenList[(dayStemIdx + 2) % 10],
    monthStem: shiShenList[(dayStemIdx + 3) % 10],
    dayStem: shiShenList[dayStemIdx], // 日主本身
    hourStem: shiShenList[(dayStemIdx + 6) % 10],
  };
}

/**
 * Create a Pillar from lunisolar char8 data.
 */
function createPillar(stem: string, branch: string): Pillar {
  const stemIdx = STEMS.indexOf(stem);
  const branchIdx = BRANCHES.indexOf(branch);
  return {
    stem,
    branch,
    stemName: stem,
    branchName: branch,
  };
}

/**
 * Main computation function — compute Bazi from birth input.
 * This is a PURE function with no side effects.
 */
export function computeBazi(input: BirthInput): BaziResult {
  // Convert input to a date string for lunisolar
  let year = input.year;
  let month = input.month;
  let day = input.day;

  // If lunar input, convert to solar first
  if (input.calendar === 'lunar') {
    const solar = toSolar(year, month, day);
    if (solar) {
      const parts = solar.split('-');
      year = parseInt(parts[0]);
      month = parseInt(parts[1]);
      day = parseInt(parts[2]);
    }
  }

  // Resolve hour
  const hour = resolveHour(input);

  // Create lunisolar instance
  const dateStr = `${year}-${String(month).padStart(2, '0')}-${String(day).padStart(2, '0')} ${String(hour).padStart(2, '0')}:00`;
  const lsr = lunisolar(dateStr);
  const char8 = lsr.char8;

  // Extract 四柱
  const yearPillar = createPillar(char8.year.stem, char8.year.branch);
  const monthPillar = createPillar(char8.month.stem, char8.month.branch);
  const dayPillar = createPillar(char8.day.stem, char8.day.branch);
  const hourPillar = createPillar(char8.hour.stem, char8.hour.branch);

  // Compute lunar date string for display
  const lunarDateStr = toLunar(year, month, day);

  // Compute solar date string for display
  const solarDateStr = `${year}-${String(month).padStart(2, '0')}-${String(day).padStart(2, '0')}`;

  // Compute 大运
  const dayun = computeDayun(year, input.gender);

  // Compute 十神 (based on day stem index)
  const dayStemIdx = STEMS.indexOf(char8.day.stem);
  const shiShen = computeShiShen(dayStemIdx);

  return {
    solarDate: solarDateStr,
    lunarDate: lunarDateStr,
    yearPillar,
    monthPillar,
    dayPillar,
    hourPillar,
    dayun,
    shiShen,
  };
}

// Import toSolar for lunar conversion (needed above)
import { toSolar, toLunar } from './calendar';
```
